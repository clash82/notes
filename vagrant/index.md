## Login to shell:  ##

{% highlight bash %}
vagrant ssh
{% endhighlight %}

## Start Vagrant: ##

{% highlight bash %}
vagrant up
{% endhighlight %}

## Stop Vagrant and clear data: ##

{% highlight bash %}
vagrant destroy
{% endhighlight %}

## Stop Vagrant but leave data: ##

{% highlight bash %}
vagrant halt
{% endhighlight %}
